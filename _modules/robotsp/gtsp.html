[docs]def convert_to_tsp_graph(graph, sets, weight='weight'):
  """
  The conversion is done using the Noon-Bean transformation as described in
  :cite:`Ben-Arieh2003`.
  """
  # A tour will have n nodes.
  # For each inter-cluster edge, add a large constant M greater than the sum
  # of the n largest weights
  tmp_graph = graph.copy()
  n = len(sets) + 1
  sorted_edges = sorted(graph.edges(data=True), key=lambda x: x[2][weight])
  M = np.sum([e[2][weight] for e in sorted_edges[-n:]])
  # M = 0
  for u,v in graph.edges_iter():
    tmp_graph.edge[u][v][weight] = graph.edge[u][v][weight] + M
  # For consecutive nodes in the same cluster, make the weights of the edges 0
  # For all other edges make the weights equal to 2M
  for members in sets:
    if len(members) < 2:
      continue
    # Consecutive nodes
    for i in xrange(len(members)):
      u = members[i]
      v = members[(i+1)%len(members)]
      tmp_graph.add_edge(u, v, weight=0)
    # All other edges
    for i in xrange(len(members)-1):
      for j in xrange(i+1, len(members)):
        u = members[i]
        v = members[j]
        if (j-i) != 1:
          tmp_graph.add_edge(u, v, weight=2*M)
  # Convert to directed
  tsp_graph = tmp_graph.to_directed()
  # Reassign the weight of each inter-cluster edge u_i,v_j to c(u_i+1,v_j)
  # Meaning c(u_next,v) <- c(u,v)
  for u,v in graph.edges_iter():
    u_next = None
    for members in sets:
      if u in members:
        if len(members) > 1:
          i = members.index(u)
          u_next = members[(i+1)%len(members)]
        break
    if u_next is not None:
      tsp_graph.edge[u_next][v][weight] = tmp_graph.edge[u][v][weight]
  return tsp_graph
